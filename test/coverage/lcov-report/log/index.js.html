/**
 * @file 框架的打点入口
 * @author sunweinan(sunweinan@baidu.com)
 *         lvlei(lvei03@baidu.com)
 */
 
/* globals swanGlobal _naSwan */
import EventsEmitter from '@baidu/events-emitter';
import PerformanceLog from './performanceLog';
import ExceptionMonitor from './exception-monitor';
import {initProductLogEvents} from './productLog';
import Statistics from './statistics';
 
export default class Log {
 
    /**
     * 日志系统构造方法
     *
     * @param {Object} global - 全局对象
     * @param {Object} swaninterface - 接口方法
     * @param {string} role - 当前环境 (master|slave)
     */
    constructor(global, swaninterface, role = 'master') {
        this.global = global;
        this.swaninterface = swaninterface;
        this.performanceLog = new PerformanceLog(swaninterface);
        this.exceptionMonitor = new ExceptionMonitor(swaninterface);
        this.statistics = new Statistics(swaninterface);
        this.swanEvents = global.swanEvents;
        this.reportFPData = false;
        this.registerLog(role);
        this.logEventsSpecialListener(role);
        Log.stageList.statistics = this.statistics;
        this.global._swanTraceList = [];
        Log.stageList.global = this.global;
    }
 
    /**
     * 注册所有类别log的监听器
     *
     * @param {string} role - 角色(master|slave)
     */
    registerLog(role) {
        // swanEvents统一在此接收
        this.swanEvents.onMessage('TraceEvents', event => {
            // 没有参数直接中断
            if (!event || !event.params) {
                return false;
            }
            const eventName = event.params.eventName || '';
            const data = event.params.data;
            this.reportMtjMiniSDK(eventName, data);
            this.executeExceptionMointor(eventName, data);
            this.executePerformPanel(eventName, data);
            this.executeStageEvent(role, eventName, data);
        });
    }
 
    /**
     * 各阶段事件处理环境
     *
     * @param {string} role - 角色(master|slave)
     * @param {string} eventName 框架阶段事件名
     * @param {*} data 通过事件传递的数据
     */
    executeStageEvent(role, eventName, data) {
        role === 'master'
        ? this.master(eventName, data)
        : this.slave(eventName, data);
    }
 
    // 内存中维护需要上报给ubcflow的list
    static perfLogList = {
        // 存放的最终上传给端上的日志的数据
        __list: [],
        add(name) {
            if ((!Log.logMessageSended)
                && ~this.__list.findIndex(item => item && item.actionId === name)) {
                return;
            }
 
            if (!Log.logMessageSended || Log.logOnNewScheme === true) {
                this.__list.push({
                    actionId: name,
                    timestamp: Date.now()
                });
            }
        },
 
        // 清空master之前的日志
        clearLog() {
            this.__list = [];
        }
    };
 
    /**
     * 稳定性打点所需要的阶段数据【TODO: 新方式收敛一段时间, 再去掉旧方案】
     */
    static stageList = {
        list: [],
        add(name) {
            if (!Log.logMessageSended) {
                if (~this.list.findIndex(item => item && item.actionId === name)) {
                    return false;
                }
                this.list.push({
                    actionId: name,
                    timestamp: Date.now()
                });
            }
        },
        send() {
            if (!Log.logMessageSended) {
                this.statistics.sendStageMessage(this.list);
                this.list = [];
            }
        },
        // 对于阶段信息的收藏, 便于客户端主动提取
        store(name) {
            this.global._swanTraceList.push({
                actionId: name,
                timestamp: Date.now() + ''
            });
        }
    };
 
    // 只首次打开上报一次
    static logMessageSended = false;
 
    // 标记本次日志发送是否是因为一个新的scheme而产生的
    static logOnNewScheme = false;
 
    /**
     * 百度统计的log
     *
     * @param {string} eventName - 事件名
     */
    reportMtjMiniSDK(eventName) {
        if (eventName === 'masterPreloadContextDecorated') {
            const {
                pageLifeCycleEventEmitter,
                appLifeCycleEventEmitter,
                swanEventsCommunicator
            } = this.global.masterManager;
            const logEventsCommunicator = EventsEmitter.merge(
                this.swanEvents,
                swanEventsCommunicator,
                pageLifeCycleEventEmitter,
                appLifeCycleEventEmitter
            );
            initProductLogEvents(logEventsCommunicator, this.swaninterface);
        }
    }
 
    /**
     * 框架流程异常事件处理
     *
     * @param {string} eventName - 事件名
     * @param {*} data - 传递的数据
     */
    executeExceptionMointor(eventName, data) {
        // 阶段名区分
        let stage;
        switch (true) {
            case eventName.includes('masterPreload'): {
                stage = 1;
                break;
            }
            case eventName.includes('masterActive'): {
                stage = 3;
                break;
            }
            case eventName.includes('slavePreload'): {
                stage = 2;
                break;
            }
            case eventName.includes('slaveActive'): {
                stage = 4;
                break;
            }
            default:
        }
 
        // 各阶段上报时机
        switch (eventName) {
            case 'masterActiveCreatePageFlowStart': {
                this.exceptionMonitor.setUserTrack(data.uri);
                break;
            }
            case 'masterPreloadStart': {
                this.exceptionMonitor.delayToSendExceptionStage(1);
                break;
            }
            case 'slavePreloadStart': {
                this.exceptionMonitor.delayToSendExceptionStage(2);
                break;
            }
            case 'masterActiveStart': {
                this.exceptionMonitor.delayToSendExceptionStage(3);
                break;
            }
            case 'slaveActiveStart': {
                // 兼容旧版swanjs
                this.global.pageRender && this.exceptionMonitor.delayToSendExceptionStage(4);
                break;
            }
        }
        stage && this.exceptionMonitor.collectEvents(stage, eventName);
    }
 
    /**
     * 客户端记录log处理
     *
     * @param {string} eventName - 事件名
     * @param {*} data - 触发事件传递的数据
     */
    executePerformPanel(eventName, data) {
        // 拿到端上关于是否显示性能面板开关的环境变量
        let showPerformancePanel = global.showPerformancePanel;
        // 如果全局有标记位并且不为true，就说明端上的环境变量传递成功且为false
        if (showPerformancePanel !== undefined && showPerformancePanel !== 'true') {
            return;
        }
        const addPerformPanel = type => {
            const perfAudit = this.global.masterManager.perfAudit;
            perfAudit[type] = parseInt(data.timestamp, 10);
            this.performanceLog.addPerformPanel([{
                slaveId: data.slaveId,
                actionName: eventName,
                timestamp: data.timestamp
            }]);
        };
 
        const performPanelEvents = {
            slaveActiveRenderEnd() {
                this.performanceLog.performPanelHelper(
                    data.slaveId,
                    this.global.pageInitRenderStart,
                    'pageInitRender'
                );
            },
            pageSwitchStart() {
                addPerformPanel('slave_navigator_start');
            },
            pageSwitchEnd() {
                addPerformPanel('slave_navigator_rendered');
            },
            pageDataUpdate() {
                this.performanceLog.performPanelHelper(data.slaveId, data.timestamp);
            }
        };
 
        // 反射客户端记录log处理事件
        try {
            performPanelEvents[eventName] && performPanelEvents[eventName].call(this);
        } catch (err) {
            console.error(`execute performPanelEvents event '${eventName}' error!`);
        }
    }
 
    /**
     * 阶段性事件处理(master), 各阶段事件处理在此扩展
     *
     * @param {string} eventName - 事件名

    master(eventName) {
        const masterEventsMap = {
            masterPreloadEnd() {

            masterActiveStart() {

            },
            masterActiveSendInitdataEnd() {

            masterOnAppLaunchHookStart() {

            // 热启动的情况下，相同的小程序被不同scheme启动
            // 在小程序被拉回的时候，实际上页面是没有被重新渲染过的，这里发送日志的终点方便统计
            masterFePageShow() {
                // 该值表示是从后台无刷新的拉回，即热启动不刷新的情况
                Log.perfLogList.add('fe_page_show');
                this.performanceLog.sendLogMessage(Log.perfLogList.__list || []);
                Log.logOnNewScheme = false;
            },
            masterOnNewScheme() {
                // 该值表示是从后台通过一个新的scheme拉回，即热启动的情况
                Log.perfLogList.clearLog();

                Log.perfLogList.add('fe_master_relaunch_start');
            },
            customComponentStatistics() {
                this.statistics.customComponentStatistics(this.global.masterManager.pagesQueue,
                    this.global.masterManager.virtualComponentFactory);
            }
        };
 
        // 反射框架master事件
        try {
            masterEventsMap[eventName] && masterEventsMap[eventName].call(this);
        } catch (err) {
            console.error(`execute master event '${eventName}' error! ${err}`);
        }
    }

                });
                let ua = this.global.navigator.userAgent;
                if (ua.match(/Android|Baidu; P1/g)) {
                    this.__fpSend = false;
                    setTimeout(() => {
                        if (!this.__fpSend) {
                            this.__fpSend = true;
                            this.sendDataToMaster('sendLogMessage', {

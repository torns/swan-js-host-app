/**
 * @file 监听框架中各个阶段的事件，上报异常case
 * @author sunweinan(sunweinan@baidu.com)
 *         lvlei(lvlei03@baidu.com)
 */
 
export default class ExceptionMonitor {
 
    constructor(swaninterface) {
        this.swaninterface = swaninterface;
    }
 
    // 管理master和slave每个点的发送状态
    static sendStatus = {
        other: 0,
        masterBeforeReady: 0,
        slaveBeforeReady: 0,
        masterAfterReady: 0,
        slaveAfterReady: 0
    };
 
    /**
     * 通过stageMap的index找到对应stage
     * 作用: 事件类型统一管理，同时简化使用
     *
     * 阶段               | 解释
     * ---------------------------------------------------------
     * other             | 预留字段
     * masterBeforeReady | master层在预加载阶段事件缺失时上报
     * slaveBeforeReady  | slave层在预加载阶段事件缺失时上报
     * masterAfterReady  | master层在用户进入小程序后的事件缺失时上报
     * slaveAfterReady   | slave层在用户进入小程序后的事件缺失时上报
     */
    static stageMap = [
        'other',
        'masterBeforeReady',
        'slaveBeforeReady',
        'masterAfterReady',
        'slaveAfterReady'
    ];
 
    /**
     *
     * stage与bizId的map
     * 作用：对应新的打点规范，特增加了与type一一对应的bizId
     * 28:other(预留字段)
     * 24:masterBeforeReady(master层在预加载阶段事件缺失时上报)
     * 25:slaveBeforeReady(slave层在预加载阶段事件缺失时上报)
     * 26:masterAfterReady(master层在用户进入小程序后的事件缺失时上报)
     * 27:slaveAfterReady(slave层在用户进入小程序后的事件缺失时上报)
     */
    static bizIdMap = [
        '28',
        '24',
        '25',
        '26',
        '27'
    ];
 
    /**
	 * 预设好的 events 组合，判断是否上报异常时将与此组合比对，上报时将转换成stage阶段码上传，节省体积
	 */
    static targetEvents = [
        [],
        [
            'masterPreloadStart',
            'masterPreloadCreateHistorystack',
            'masterPreloadCommunicatorListened',
            'masterPreloadVirtualComponentFactoryInstantiated',
            'masterPreloadExtensionInstantiated',
            'masterPreloadAllSlaveEventsListened',
            'masterPreloadMountMastermanagerToGlobal',
            'masterPreloadContextDecorated',
            'masterPreloadAppReadyListened',
            'masterPreloadEnd'
        ],
        [
            'slavePreloadStart',
            'slavePreloadGetComponents',
            'slavePreloadGetComponentFactory',
            'slavePreloadDefineComponentsStart',
            'slavePreloadDefineComponentsEnd',
            'slavePreloadPageReadyListened',
            'slavePreloadEnd'
        ],
 
        [
            'masterActiveStart',
            'masterActiveInitRenderStart',
            'masterActivePushInitSlaveStart',
            'masterActiveCreateInitSlave',
            'masterActiveInitAction',
            'masterActiveInitBindingEnvironmentEvents',
            'masterActiveAppJsLoaded',
            'masterActiveCreateInitSlaveEnd',
            'masterActivePushInitSlaveEnd',
            'masterActiveCreatePageFlowStart',
            'masterActiveGetUserPageInstance',
            'masterActiveInitUserPageInstance',
            'masterActiveCreatePageFlowEnd',
            'masterActiveSendInitdataStart',
            'masterActiveSendInitdataEnd',
            'masterActiveOnqueueInitSlave'
        ],
        [
            'slaveActiveStart',
            'slaveActivePageLoadStart',
            'slaveActiveCssLoaded',
            'slaveActiveSwanJsLoadStart',
            'slaveActivePageRender',
            'slaveActiveDefinePageComponentEnd',

            'slaveActiveNoticeMasterSlaveloaded',
            'slaveActiveJsParsed',
            'slaveActiveSwanJsLoaded',
            'slaveActiveReceiveInitData',
            'slaveActiveRenderStart',
            'slaveActiveRenderEnd',
            'slaveActivePageAttached'
        ]
    ];
 
    /**
     * 当前收集到的 eventlist
     * 会在代码中各个部份来 pushevent
     */
    static currentEvents = [
        [],
        [],
        [],
        [],
        []
    ];
 
    /**
     * 用户点击轨迹list
     */
    static userTrack = [];
 
    /**
     * 收集事件，更新currentEvents
     *
     * @param  {number} index - 对应stageMap的异常阶段序号
     * @param  {string} name - 每个阶段具体的节点名
     * @param  {Objcet} data 详细数据，考虑到传输大小，暂时不开放这个功能
     */
    collectEvents(index, name, data) {
        const [targetEvents, currentEvents] = [
            ExceptionMonitor.targetEvents,
            ExceptionMonitor.currentEvents
        ].map(events => events[index].filter(event => event === name));
 
        if (targetEvents.length && !currentEvents.length) {
            ExceptionMonitor.currentEvents[index].push(targetEvents[0]);
        }
    }
 
    /**
     * 检查已收集的事件是否符合预设，作为是否上报异常的依据
     *
     * @param  {string} index - 对应stageMap的异常阶段序号
     * @return {boolean} 是否符合预设
     */
    isValid(index) {
        return ExceptionMonitor.targetEvents[index].length === ExceptionMonitor.currentEvents[index].length;
    }
 
    /**
     * 找出丢失的点
     *
     * @param {number} index - 对应stageMap的异常阶段序号
     * @param {Array} targetEvents - 预设好的 events 组合
     * @param {Array} currentEvents - 当前的预 events 组合
     * @return {Array} - 丢失的点
     */
    getLost(index, targetEvents, currentEvents) {
        const lost = targetEvents[index]
            .filter(targetEvent => !currentEvents[index].includes(targetEvent));
        return lost.map(stage => {
            const lostIndex = targetEvents[index]
                .findIndex(targetEvent => targetEvent === stage);
            return {
                index: `${index}_${lostIndex}`,
                stage
            };
        });
    }
 
    /**
     * 找出丢失的点
     *
     * @param {number} index - 对应stageMap的异常阶段序号
     * @param {Array} targetEvents - 预设好的 events 组合
     * @param {Array} currentEvents - 当前的预 events 组合
     * @return {string} - 上报阶段代替名
     */
    getRest(index, targetEvents, currentEvents) {
        return currentEvents[index]
            .map(currentEvent => {
                const stage = targetEvents[index]
                    .findIndex(targetEvent => targetEvent === currentEvent);
                return `${index}_${stage}`;
            });
    }
 
    /**
     * 开始捕捉某种类型的异常
     *
     * @param {number} index - 对应stageMap的异常阶段序号
     * @param {number} time - 等待的时间，该时间后将开始比对异常并上报
     */
    delayToSendExceptionStage(index, time = 5000) {
        setTimeout(() => {
            const stageName = ExceptionMonitor.stageMap[index];
            const bizId = ExceptionMonitor.bizIdMap[index];
            const [targetEvents, currentEvents]
                = [ExceptionMonitor.targetEvents, ExceptionMonitor.currentEvents];
            const pass = this.isValid(index);
            const rest = this.getRest(index, targetEvents, currentEvents);
            const lost = this.getLost(index, targetEvents, currentEvents);
            const stageInfo = {
                pass,
                rest,
                lost
            };
            if (!ExceptionMonitor.sendStatus[stageName]) {
                this.send(bizId, stageName, stageInfo);
                ExceptionMonitor.sendStatus[stageName] = 1;
            }
        }, time);
    }
 
    /**
     * 设置用户点击轨迹list，定长，超出后弹出
     *
     * @param  {string} uri - page路径
     */
    setUserTrack(uri) {
        if (ExceptionMonitor.userTrack.length >= 10) {
            ExceptionMonitor.userTrack.pop();
        }
        ExceptionMonitor.userTrack.push(uri);
    }
 
    /**
     * 获取系统信息
     *
     * @return {Object} 返回系统信息
     */
    getSystemInfo() {
        const systemInfo = this.swaninterface.swan.getSystemInfoSync();
        return {
            SDKVersion: systemInfo.SDKVersion,
            model: systemInfo.model,
            system: systemInfo.system,
            version: systemInfo.version
        };
    }
 
    /**
     * 获取app信息
     *
     * @return {Object} 返回小程序信息
     */
    getAppInfo() {
        const appInfo = this.swaninterface.boxjs.data.get({
            name: 'swan-appInfoSync'
        }) || {};
        return {
            cuid: appInfo.cuid,
            appid: appInfo.appid
        };
    }
 
    /**
     * 将打点数据附加一些基础信息
     *
     * @return {Object} 附加上额外信息
     */
    extendExtInfo() {
        try {
            const systemInfo = this.getSystemInfo();
            const appInfo = this.getAppInfo();
            const userTrack = ExceptionMonitor.userTrack;
            return {
                systemInfo,
                appInfo,
                userTrack
            };
        } catch (e) {
            console.error(e);
            return {};
        }
    }
 
    /**
     * 具体上报逻辑
     *
     * @param  {string} bizId - bizId
     * @param  {string} stageName - 上传阶段名
     * @param  {Object} stageInfo - 该阶段剩余点以及丢失点的信息
     */
    send(bizId, stageName, stageInfo = {}) {
        try {
            const baseInfo = this.extendExtInfo();
            // ubc打点
            this.swaninterface.boxjs.log({
                name: 'ubcReport',
                data: {
                    actionId: '767',
                    value: {
                        from: 'swan',
                        type: stageName,
                        ext: {
                            stageInfo,
                            ...baseInfo
                        }
                    }
                }
            });
 
            // 双打767
            this.swaninterface.boxjs.log({
                name: 'openStatisticEvent',
                data: {
                    groupId: '9',
                    bizId,
                    eventName: stageName,
                    content: {
                        ext: {
                            stageInfo,
                            ...baseInfo
                        }
                    }
                }
            });
        } catch (err) {
            console.error(`ubcReport_767 error: ${err}`);
        }
    }
}

 /**
 * 数组去重
 *
 * @param {Array} originArray -去重的数组
 * @return {Array} temp -去重后的数组
 */
export const uniqArray = originArray => {
    let temp = [];
    let l = originArray.length;
    for (let i = 0; i < l; i++) {
        for (let j = i + 1; j < l; j++) {
            if (originArray[i] === originArray[j]) {
                i++;
                j = i;
            }
        }
        temp.push(originArray[i]);
    }
    return temp;
};
